# Phase 4.1 / 4.2 — vector extraction dependencies
# Install into a venv before running extract_vectors.py:
#   python -m venv .venv && source .venv/bin/activate && pip install -r requirements.txt

# Core ML stack
# NB: Installing this file alone may pull a default PyPI torch wheel whose
# bundled CUDA doesn't match the host driver (RunPod's CUDA-12 boxes get
# cu13 wheels by default → torch.cuda.is_available() == False). Use
# scripts/setup_runpod.sh, which detects host CUDA and pre-installs torch
# from the matching pytorch.org wheel channel (cu126/cu128/etc.) before
# resolving this file. Manual install: `pip install --index-url
# https://download.pytorch.org/whl/cu126 torch torchvision torchaudio`
# first, then `pip install -r requirements.txt`.
torch>=2.3,<3            # PyTorch; CUDA/MPS/CPU backends
transformers>=4.50,<5    # HF transformers with Qwen3 support (4.50+ required)
accelerate               # device_map + multi-GPU / offload support

# Steering vector library
repeng>=0.3              # control-vector extraction; 0.3.x is the stable series

# Numerics / data
numpy                    # array ops, npz I/O

# Spec parsing
pyyaml                   # YAML spec loading (pure Python, no C extension required)

# Version notes:
#   transformers 4.50+ is required for Qwen3-32B tokenizer + model class support.
#   repeng 0.3.x exposes ControlVector and DatasetEntry; earlier versions have
#   a different API.  The extraction utility uses transformers hooks directly
#   (register_forward_hook) and only uses repeng for its dataset utilities and
#   version identification.
#   torch 2.3+ is required for bfloat16 MPS support on Apple Silicon.
#   accelerate is needed for device_map="auto" / "mps" with large models.
#   numpy 1.x and 2.x are both compatible; 2.x preferred for fp8 array support.
