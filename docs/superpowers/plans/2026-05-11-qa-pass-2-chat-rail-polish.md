### Task 1: Fix ModelPicker React key collision (QA item 2.5)

The current key uses only `o.model`, so when two providers expose the same model name (e.g., `claude-sonnet-4-6` on Anthropic and OpenRouter, or `deepseek-v4-pro` on both DeepSeek native and OpenRouter), React keys collide. The fix is one line. Per the QA spec, we do NOT dedupe or normalize — we keep both options visible.

**Files:**
- Modify: `frontend/web/src/components/shell/ChatRail.tsx:454-462`

- [ ] **Step 1: Apply the key fix**

Locate the `<option>` render inside `ModelPicker`:

```tsx
{groupByProvider(options).map((g) => (
  <optgroup key={g.provider} label={g.provider}>
    {g.items.map((o) => (
      <option key={o.model} value={`${o.provider}::${o.model}`}>
        {o.model}
      </option>
    ))}
  </optgroup>
))}
```

Change the inner `key`:

```tsx
{groupByProvider(options).map((g) => (
  <optgroup key={g.provider} label={g.provider}>
    {g.items.map((o) => (
      <option key={`${o.provider}::${o.model}`} value={`${o.provider}::${o.model}`}>
        {o.model}
      </option>
    ))}
  </optgroup>
))}
```

No other code changes. The `<optgroup label={g.provider}>` already gives the user the provider context visually, so the option label stays `{o.model}` — adding provider to the label would be redundant.

### Task 4: Inline confirmations for `xvn` tool calls (QA item 2.3)

Today tool calls render as opaque pills (`create_strategy · 01KRBS2FR61F91HZ57AP62RV0W`). Add a human-readable narrative line above the next assistant token so the user can read what the model just did.

**Files:**
- Modify: `frontend/web/src/components/shell/ChatRail.tsx` — add `toolNarrative()` helper near `summarizeResult` (line ~685), update bubble render in `Thread`

- [ ] **Step 1: Add `toolNarrative()` helper**

Add immediately after `summarizeResult` (find the existing function around line 670-699):

```tsx
type ToolEntry = {
  call: string;
  args?: unknown;
  result?: unknown;
};

/**
 * Map a completed tool invocation to a human-readable inline confirmation.
 * Returns null for tools whose result is best left implicit (read-only tools
 * where the model's prose already reflects the outcome).
 */
function toolNarrative(tool: ToolEntry): { ok: boolean; text: string } | null {
  if (tool.result === undefined) return null; // still running — handled by spinner
  const args = (tool.args ?? {}) as Record<string, unknown>;
  const result = (tool.result ?? {}) as Record<string, unknown>;
  const ok = !("error" in result);

  switch (tool.call) {
    case "create_strategy": {
      const name = String(args.display_name ?? args.name ?? "(unnamed)");
      const id = String(result.agent_id ?? result.id ?? "");
      return ok
        ? { ok: true, text: `Created strategy **${name}**${id ? ` (\`${id}\`)` : ""}` }
        : { ok: false, text: `Failed to create strategy: ${String(result.error ?? "unknown error")}` };
    }
    case "set_mechanical_param": {
      const param = String(args.param ?? "?");
      const value = String(args.value ?? "?");
      return ok
        ? { ok: true, text: `Set \`${param}\` = ${value}` }
        : { ok: false, text: `Failed to set \`${param}\`: ${String(result.error ?? "unknown error")}` };
    }
    case "set_risk_config": {
      const level = String(args.level ?? "");
      const perTrade = String(args.per_trade ?? args.risk_per_trade ?? "");
      const dailyKill = String(args.daily_loss_kill ?? args.daily_kill ?? "");
      const parts = [
        level && `level: ${level}`,
        perTrade && `${perTrade} per trade`,
        dailyKill && `${dailyKill} daily kill`,
      ].filter(Boolean);
      return ok
        ? { ok: true, text: `Risk: ${parts.join(", ")}` }
        : { ok: false, text: `Failed to set risk: ${String(result.error ?? "unknown error")}` };
    }
    case "validate_draft": {
      return ok
        ? { ok: true, text: "Validation passed" }
        : { ok: false, text: `Validation failed: ${String(result.error ?? "see details")}` };
    }
    case "get_strategy":
    case "list_templates":
      // Read-only — model's prose already covers it.
      return null;
    default:
      return null;
  }
}
```

```tsx
    remarkPlugins={[remarkGfm]}
    components={{
      // Inline code
      code: ({ node, inline, className, children, ...props }: any) =>
        inline ? (
          <code
            className="font-mono text-[12px] bg-surface-2/60 px-1 py-0.5 rounded"
            {...props}
          >
            {children}
          </code>
        ) : (
          <pre className="font-mono text-[12px] bg-surface-2/60 p-2 rounded my-2 overflow-x-auto">
            <code {...props}>{children}</code>
          </pre>
        ),
      // Tables
      table: ({ children }) => (
        <table className="border-collapse my-2 text-[12px]">{children}</table>
      ),
      th: ({ children }) => (
        <th className="border border-border-soft px-2 py-1 text-left font-medium">{children}</th>
      ),
      td: ({ children }) => (
        <td className="border border-border-soft px-2 py-1">{children}</td>
      ),
      // Lists
      ul: ({ children }) => <ul className="list-disc pl-5 my-1.5">{children}</ul>,
      ol: ({ children }) => <ol className="list-decimal pl-5 my-1.5">{children}</ol>,
      // Strong / em — inherit theme defaults
      strong: ({ children }) => <strong className="text-text font-semibold">{children}</strong>,
    }}
  >
    {b.text}
  </ReactMarkdown>
</div>
```

User bubbles stay as-is (plain `whitespace-pre-wrap` text) — the user types prose, not markdown.
